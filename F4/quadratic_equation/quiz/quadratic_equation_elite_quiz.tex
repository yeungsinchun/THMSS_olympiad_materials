	\question[12]
	\begin{parts} \part Let $\alpha$, $\beta$ be the roots of $x^2+px+q=0 \quad (*)$, where $p$ and $q$ are constants. Find, in terms of $p$ and $q$.
		\begin{subparts}
			\subpart[2] $\alpha^2+\beta^2$
			\subpart[2] $\alpha^3+\beta^3$
			\subpart[2] $(\alpha^2-\beta-1)(\beta^2-\alpha-1)$.
		\end{subparts}
		\part[2] If the square of one root of (*) minus the other root equals $1$, use (a), or otherwise, to show that $q^2-3(p-1)q+(p-1)^2(p+1)=0 \quad (**)$
		\part[2] Find the range of values of $p$ such that the quadratic equation $(**)$ has real roots.
		\part[2] Suppose $k$ is a real constant. If the square of one root of $4x^2+5x+k=0$ minus the other root equals $1$, use the result in $(b)$, or otherwise, to find the value of $k$.
	\end{parts}
	\question[4] The quadratic equation $9x^2-(k+1)x+1=0 \quad (*)$ has equal roots.
	\begin{parts}
		\part Find the two possible values of the constant $k$.
		\part If $k$ takes the negative value obtained, solve equation $(*)$.
	\end{parts}
	\question[4] $\alpha$ and $\beta$ are the roots of the quadratic equation $kx^2-4x+2k=0$, where $k (k \neq 0)$ is a constant. Express the following in terms of $k$.
	\question[4] The equation $x^2+4x+p=0$, where $p$ is a real constant, has distinct real roots $\alpha$ and $\beta$.
	\begin{parts}
		\part Find the range of values of $p$
		\part If $\alpha^2+\beta^2+\alpha^2\beta^2+3(\alpha+\beta)-19=0$, find the values of $p$.
	\end{parts}
